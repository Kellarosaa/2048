// Define an array of numbers with their corresponding names const elements = [ { number: 1, name: "One" }, { number: 2, name: "Two" }, { number: 3, name: "Three" }, { number: 4, name: "Four" }, { number: 5, name: "Five" } ]; // Function to craft a random number function craftNumber() { return elements[Math.floor(Math.random() * elements.length)]; } // Function to play the interactive game function playInfiniteCraftGame() { const { number, name } = craftNumber(); console.log(`Crafted Number: ${number} (${name})`); } // Play the interactive game playInfiniteCraftGame();
